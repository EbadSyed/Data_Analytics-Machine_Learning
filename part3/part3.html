Part 3.2 Wavelet

[cfs,frq] = cwt(Signal,Fs);
tms = (0:numel(Signal)-1)/Fs;
figure(2)
subplot(2,1,1)
plot(tms,Signal)
axis tight
title('Signal and Scalogram')
xlabel('Time (s)')
ylabel('Amplitude')
subplot(2,1,2)
surface(tms,frq,abs(cfs))
axis tight
shading flat
xlabel('Time (s)')
ylabel('Frequency (Hz)')
set(gca,'yscale','log')

load('signals.mat')
fs=500;
minF = 2;
maxF = 150;
generated_points = 99;
lfp1 = lfp1(1:5000);
stepSize = (maxF-minF)/generated_points;
coeffs1 = morlet_wavelet(lfp1,fs,[minF:stepSize:maxF]);
tms1 = (0:numel(lfp1)-1)/fs;
figure(3)
subplot(2,1,1)
plot(tms1,lfp1)
axis tight
title('LFP1 Signal and Scalogram')
xlabel('Time (s)')
ylabel('Amplitude')
subplot(2,1,2)
surface(tms1,[minF:stepSize:maxF],abs(coeffs1))
